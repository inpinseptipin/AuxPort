42 std::string Time::getCurrentTime(const TimeType& timeType)
43 {
44 time_t currentTime = std::time(0);
45 std::stringstream ss;
46 ss << std::put_time(std::localtime(&currentTime), "%c");
47 std::string returnTime = ss.str();
48 switch (timeType)
49 {
50 case TimeType::Day:
51 return "Day : " + returnTime.substr(0, 3);
52 break;
53
54 case TimeType::Date:
55 return "Date : " + returnTime.substr(4, 6);
56 break;
57
58 case TimeType::time:
59 return "Time : " + returnTime.substr(11, 8);
60 break;
61
62 case TimeType::Year:
63 return "Year : " + returnTime.substr(20, 4);
64 break;
65
66 case TimeType::Raw:
67 return returnTime.substr(0, 24);
68 break;
69 }
70 return returnTime;
71 }
