85 float compute(const std::vector<errorVal>& calculatedValues, const std::vector<errorVal>& expectedValues)
86 {
87 AuxAssert(calculatedValues.size() == expectedValues.size(), "Vectors must be of same size!");
88
89 float result = 0.0f;
90 size_t n = calculatedValues.size();
91
92 switch (type)
93 {
94 case Type::Mean:
95 for (size_t i = 0; i < n; i++)
96 result += (calculatedValues[i] - expectedValues[i]);
97 result /= n;
98 break;
99
100 case Type::Absolute:
101 for (size_t i = 0; i < n; i++)
102 result += std::abs(calculatedValues[i] - expectedValues[i]);
103 result /= n;
104 break;
105
106 case Type::Relative:
107 for (size_t i = 0; i < n; i++)
108 result += std::abs(calculatedValues[i] - expectedValues[i]) / std::abs(expectedValues[i]);
109 result /= n;
110 break;
111
112 case Type::RMSE:
113 for (size_t i = 0; i < n; i++)
114 result += std::norm((calculatedValues[i] - expectedValues[i]));
115 result = std::sqrt((float)result / n);
116 break;
117 }
118 return result;
119 }
120 private:
121 Type type;
122 };
