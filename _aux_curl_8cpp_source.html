51bool AuxPort::Extensions::AuxCurl::sendRequest(const RequestType& type)
52{
53 switch (type)
54 {
55 case GET:
56 curl_easy_setopt(curlHandle, CURLOPT_HTTPGET, 1L);
57 break;
58 case POST:
59 curl_easy_setopt(curlHandle, CURLOPT_POST, 1L);
60 break;
61 }
62
63 // Making sure responseData does not contain previous responses
64 responseData = "";
65 CURLcode responseCode = curl_easy_perform(curlHandle);
66 if (responseCode != CURLE_OK)
67 {
68 responseData = "Error: ";
69 responseData += curl_easy_strerror(responseCode);
70 responseHandler(responseData);
71 return false;
72 }
73 responseHandler(responseData);
74 return true;
75}
