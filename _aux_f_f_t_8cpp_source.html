48void AuxPort::Audio::FourierTransform::compute()
49{
50 std::complex<float> temp;
51 uint32_t N = static_cast<uint32_t>(_fftValues.size());
52 uint32_t K = N;
53 uint32_t n;
54 float theta = static_cast<float>(pi / N);
55 std::complex<float> phi = std::complex<float>(cos(theta), -sin(theta));
56 std::complex<float> T = 1.0L;
57
58 while (K > 0)
59 {
60 n = K;
61 K >>= 1;
62 phi = phi * phi;
63 T = 1.0L;
64 for (uint32_t i = 0; i < K; i++)
65 {
66 for (uint32_t j = i; j < N; j += n)
67 {
68 uint32_t b = j + K;
69 temp = _fftValues[j] - _fftValues[b];
70 _fftValues[j] += _fftValues[b];
71 _fftValues[b] = temp * T;
72 }
73 T *= phi;
74 }
75 }
76 int32_t numberOfBits = (int32_t)log2(N);
77 for (uint32_t i = 0; i < N; i++)
78 {
79 uint32_t j = reverseBits(i);
80 if (j > i)
81 {
82 temp = _fftValues[i];
83 _fftValues[i] = _fftValues[j];
84 _fftValues[j] = temp;
85 }
86 }
87}

89uint32 AuxPort::Audio::FourierTransform::reverseBits(uint32 b)
90{
91 b = (((b & 0xaaaaaaaa) >> 1) | ((b & 0x55555555) << 1));
92 b = (((b & 0xcccccccc) >> 2) | ((b & 0x33333333) << 2));
93 b = (((b & 0xf0f0f0f0) >> 4) | ((b & 0x0f0f0f0f) << 4));
94 b = (((b & 0xff00ff00) >> 8) | ((b & 0x00ff00ff) << 8));
95 b = ((b >> 16) | (b << 16)) >> (32 - _log2N);
96 return b;
97}

162void AuxPort::Audio::DiscreteCosineTransform::computeTransform(const std::vector<float>& inputBuffer, std::vector<float>& outputBuffer)
163{
164 AuxAssert(inputBuffer.size() == _dctValues.size(), "Size of input buffer not same as the DCT Engine's expectation");
165 AuxAssert(outputBuffer.size() == inputBuffer.size(), "DCT output buffer's size should be the same as the input audio buffer");
166
167 uint32_t N = inputBuffer.size();
168 float theta;
169 float scalingFactor = sqrt(2.0f / N);
170 for (uint32_t i = 0; i < N; i++)
171 {
172 _dctValues[i] = 0;
173 for (uint32_t j = 0; j < N; j++)
174 {
175 theta = AuxPort::pi * (2 * j + 1) * i / (2 * N);
176 _dctValues[i] += inputBuffer[j] * cos(theta);
177 }
178 _dctValues[i] *= scalingFactor;
179 }
180 _dctValues[0] *= 0.707106781186f;
181
182 for (uint32_t i = 0; i < N; i++)
183 outputBuffer[i] = _dctValues[i];
