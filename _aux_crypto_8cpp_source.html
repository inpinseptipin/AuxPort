55std::vector<uint8_t> AuxPort::MD5::getPaddedData(const std::string& data)
56{
57 std::vector<uint8_t> paddedData(data.begin(), data.end());
58 paddedData.push_back(0x80);
59 while (paddedData.size() % 64 != 56)
60 paddedData.push_back(0x00);
61
62 // Padding with original data size
63 size_t dataLengthInBits = data.size() * 8;
64 for (int i = 0; i < 8; ++i)
65 {
66 paddedData.push_back(dataLengthInBits & 0xFF);
67 dataLengthInBits >>= 8;
68 }
69 return paddedData;
70}

72std::string AuxPort::MD5::computeHash(const std::vector<uint8_t>& paddedData)
73{
74 uint32_t a0 = 0x67452301;
75 uint32_t b0 = 0xefcdab89;
76 uint32_t c0 = 0x98badcfe;
77 uint32_t d0 = 0x10325476;
78
79 size_t blockCount = paddedData.size() / 64;
80 for (size_t blockIndex = 0; blockIndex < blockCount; ++blockIndex)
81 {
82 uint32_t AA = a0;
83 uint32_t BB = b0;
84 uint32_t CC = c0;
85 uint32_t DD = d0;
86
87 uint32_t E;
88 unsigned int j;
89 for (unsigned int i = 0; i < 64; ++i)
90 {
91 switch (i / 16)
92 {
93 case 0:
94 E = F(BB, CC, DD);
95 j = i;
96 break;
97 case 1:
98 E = G(BB, CC, DD);
99 j = ((i * 5) + 1) % 16;
100 break;
101 case 2:
102 E = H(BB, CC, DD);
103 j = ((i * 3) + 5) % 16;
104 break;
105 default:
106 E = I(BB, CC, DD);
107 j = (i * 7) % 16;
108 break;
109 }
110
111 uint32_t temp = DD;
112 DD = CC;
113 CC = BB;
114 BB = BB + rotateLeft(AA + E + K[i] + getWord(paddedData, blockIndex * 16 + j), S[i]);
115 AA = temp;
116 }
117
118 a0 += AA;
119 b0 += BB;
120 c0 += CC;
121 d0 += DD;
122 }

125 b0 = convertToBigEndian(b0);
126 c0 = convertToBigEndian(c0);
