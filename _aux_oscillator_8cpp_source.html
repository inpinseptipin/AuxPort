194float AuxPort::Audio::ADSR::process()
195{
196 if (this->state == State::Attack)
197 {
198 envelope += mods[Attack];
199 if (envelope >= parameters[Attack])
200 this->state = State::Decay;
201 }
202 if (this->state == State::Decay)
203 {
204 envelope -= mods[Decay];
205 if (envelope <= parameters[Decay])
206 this->state = State::Sustain;
207 }
208 if (this->state == State::Sustain)
209 {
210 envelope -= mods[Sustain];
211 count++;
212 if (envelope < parameters[Sustain])
213 envelope = parameters[Sustain];
214 if (count >= parameters[SustainTime] * sampleRate)
215 {
216 count = 0;
217 this->state = State::Release;
218 }
219 }
220 if (this->state == State::Release)
221 {
222 envelope -= mods[Release];
223 if (envelope <= 0)
224 {
225 this->state = Attack;
226 }
227
228 }
229 return envelope;
230}
