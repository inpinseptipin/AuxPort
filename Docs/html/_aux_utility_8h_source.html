234 {
235 AuxAssert(vector.size() > 0, "Vector size should be greater than 0");
236 if (vector.size() == 1)
237 return vector[0];
238 sample max = vector[0];
239 for (uint32_t i = 1; i < vector.size(); i++)
240 max = max < vector[i] ? vector[i] : max;
241 return max;

298 static void split(std::vector<std::string>& tokens, const std::string& str, const char& delimiter)
299 {
300 tokens.clear();
301 std::stringstream stream(str);
302 std::string token;

312 static void join(std::string& str, const std::vector<std::string>& tokens, const char& delimiter)
313 {
314 str = "";
315 if (tokens.empty()) return;
316 size_t tokensCount = tokens.size();
317 for (size_t i = 0; i < tokensCount - 1; i++)
318 {
319 str += tokens[i] + delimiter;
320 }
321 str += tokens[tokensCount - 1];
322 }
