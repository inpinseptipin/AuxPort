99float AuxPort::Audio::Distortion::schetzenOverdrive(float& audio, const bool& preserve)
100{
101 auto x = abs(audio);
102 if (preserve)
103 {
104 if (x >= 0.0f && x < 0.34f)
105 return 2 * audio;
106 else if (x >= 0.34f && x < 0.66f)
107 return audio < 0 ? -(3 - powf(2 - 3 * audio, 2)) / 3 : (3 - powf(2 - 3 * audio, 2)) / 3;
108 else if (x >= 0.66f && x <= 1.0f)
109 return audio < 0 ? -1.0f : 1.0f;
110 }
111 else
112 {
113 if (x >= 0.0f && x < 0.34f)
114 audio = 2 * audio;
115 else if (x >= 0.34f && x < 0.66f)
116 audio = audio < 0 ? -(3 - powf(2 - 3 * audio, 2)) / 3 : (3 - powf(2 - 3 * audio, 2)) / 3;
117 else if (x >= 0.66f && x <= 1.0f)
118 audio = audio < 0 ? -1.0f : 1.0f;
119 }
120 return 0.0f;
121
122}

152float AuxPort::Audio::Distortion::levianTanh(float& audio, const bool& preserve)
153{
154 float x2 = audio * audio;
155 float x3 = x2 * audio;
156 float x5 = x2 * x3;
157 float z = audio + (0.16489087f * x3) + (0.00985468f * x5);
158 return preserve == true ? audio = z / sqrtf(1.f + (z * z)) : audio = z / sqrtf(1.f + (z * z));
159}

161float AuxPort::Audio::Assymetric::process(float sample,float k,float g = 1.0,Type type = Type::v1)
162{
163 if (type == v1)
164 {
165 if (sample >= 0)
166 circBuffer.push(Distortion::hyperbolicTan(sample, k, true));
167 if (sample < 0)
168 circBuffer.push(Distortion::arcTan(sample,k,true));
169 }
170 if (type == v2)
171 {
172 if (sample >= 0)
173 circBuffer.push(Distortion::arcTan(sample, k, true) / Distortion::arcTan(k, 1, true));
174 if (sample < 0)
175 {
176 auto constant = k / g;
177 circBuffer.push((g * Distortion::arcTan(sample, k / g, true)) / Distortion::arcTan(constant, 1.0, true));
178 }
179 }
180 counter++;
181 if (counter == circBuffer.getSize() - 1)
182 circBuffer.meanRemoval();
183 return circBuffer.pop();
184}
