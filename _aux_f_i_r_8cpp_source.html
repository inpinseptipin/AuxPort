60void AuxPort::Audio::FIR::compute(float passband, float stopband, uint32_t order, Type filterType)
61{
62 AuxAssert(this->sampleRate > 0, "Call setSampleRate() before calling compute");
63 AuxAssert(stopband > 0 && passband > 0, "Passband and Stopband should be greater than 0");
64 AuxAssert(stopband > passband, "Stopband should be greater than the passband");
65 AuxAssert(order < 1000, "Why are you creating a 1000 tap FIR?... What's wrong with you");
66 this->passband = (2 * AuxPort::pi * passband) / this->sampleRate;
67 this->stopband = (2 * AuxPort::pi * stopband) / this->sampleRate;
68 this->cutoff = (this->passband + this->stopband) / 2;
69 this->normFreq = this->cutoff;
70 order = order % 2 == 1 ? order : order+1;
71 this->order = order;
72 impulseResponse.resize(order);
73 this->filterType = filterType;
74
75 filterAlgo();
76 applyWindow();
77 normalize();
78}
